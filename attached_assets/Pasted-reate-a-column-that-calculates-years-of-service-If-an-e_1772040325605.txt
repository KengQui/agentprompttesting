from datetime import datetime

def calculate_years_of_service(date_hired, date_re_hired):
    today = datetime(2026, 2, 25)
    years_of_service = ''
    if date_re_hired:
        time_difference = today - date_re_hired
        years = time_difference.days / 365.25
    else:
        time_difference = today - date_hired
        years = time_difference.days / 365.25
    
    if years >= 5:
        years_of_service = str(round(years, 2))
    
    return years_of_service
